    Fix camera-orbit hang and expand world detail
    
    Fix render-loop crash: arrow-key camera orbit called controls.rotateLeft/
    rotateUp, which are not public methods in three.js r128. The thrown
    TypeError aborted animate() before setAnimationLoop could schedule the next
    frame, permanently freezing rendering and WASD on the first arrow-key press.
    Replace with an orbitCamera() helper using THREE.Spherical, respecting the
    existing polar-angle limits. Verified in-browser: orbiting + walking now
    sustains continuous draw calls with zero errors.
    
    Also lands in-progress world work: rolling-terrain detail, boulders, flowers
    and grass tufts, classroom assets, door frames, exterior signage, theater
    mode, and ambient NPCs.

+function getTerrainHeight(x, z) {
+  const distFromCenter = Math.sqrt(x * x + z * z);
+  if (distFromCenter < 52) {
+    return 0; // Flat safety zone for the building foundation
+  }
+  // Smoothly blend hills into flat ground near the building
+  const blendFactor = Math.min((distFromCenter - 52) * 0.08, 1.0);
+  const hills = Math.sin(x * 0.1) * Math.cos(z * 0.1) * 2.2 + Math.cos(x * 0.05) * Math.sin(z * 0.05) * 1.5;
+  return hills * blendFactor;
+}
